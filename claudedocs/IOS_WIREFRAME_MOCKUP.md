```swift
struct StatGridView: View {
    struct StatItem: Identifiable {
        let id = UUID()
        let label: String
        let value: String
        let trend: Trend?

        enum Trend {
            case up, down, neutral

            var icon: String {
                switch self {
                case .up: return "arrow.up"
                case .down: return "arrow.down"
                case .neutral: return "minus"
                }
            }

            var color: Color {
                switch self {
                case .up: return .green
                case .down: return .red
                case .neutral: return .gray
                }
            }
        }
    }

    let items: [StatItem]
    var columns: Int = 2

    var body: some View {
        LazyVGrid(columns: Array(repeating: GridItem(.flexible()), count: columns), spacing: 16) {
            ForEach(items) { item in
                VStack(alignment: .leading, spacing: 4) {
                    Text(item.label)
                        .font(.caption)
                        .foregroundColor(.secondary)

                    HStack(spacing: 4) {
                        Text(item.value)
                            .font(.system(size: 24, weight: .bold, design: .monospaced))

                        if let trend = item.trend {
                            Image(systemName: trend.icon)
                                .font(.caption)
                                .foregroundColor(trend.color)
                        }
                    }
                }
                .frame(maxWidth: .infinity, alignment: .leading)
                .padding()
                .background(.ultraThinMaterial)
                .clipShape(RoundedRectangle(cornerRadius: 12))
            }
        }
    }
```

```swift
// Repository Pattern
class GamesRepository: ObservableObject {
    @Published var games: [Game] = []
    @Published var isLoading = false

    private let modelContext: ModelContext

    func loadGames() async {
        // 1. Load from cache immediately
        let cached = try? modelContext.fetch(FetchDescriptor<CachedGame>())
        if let cached = cached {
            await MainActor.run {
                self.games = cached.map { $0.toGame() }
            }
        }

        // 2. Fetch fresh data in background
        await MainActor.run { isLoading = true }
        do {
            let freshGames = try await APIClient.shared.fetchTodayGames()

            // 3. Update cache
            for game in freshGames {
                let cached = CachedGame(from: game)
                modelContext.insert(cached)
            }
            try modelContext.save()

            // 4. Update UI
            await MainActor.run {
                self.games = freshGames
                self.isLoading = false
            }
        } catch {
            await MainActor.run { isLoading = false }
        }
    }
}
```

### 8.1 Monte Carlo Distribution Chart

```swift
import Charts

struct MonteCarloDistributionChart: View {
    let result: MonteCarloResult
    let line: Double

    var body: some View {
        Chart {
            // 95% Confidence Interval Area
            AreaMark(
                x: .value("Total", result.percentiles.p5),
                yStart: .value("Probability", 0),
                yEnd: .value("Probability", 0.05)
            )
            .foregroundStyle(
                .linearGradient(
                    colors: [.blue.opacity(0.1), .blue.opacity(0.3)],
                    startPoint: .bottom,
                    endPoint: .top
                )
            )

            // Distribution curve approximation
            ForEach(distributionPoints, id: \.total) { point in
                LineMark(
                    x: .value("Total", point.total),
                    y: .value("Probability", point.probability)
                )
                .foregroundStyle(.blue)
                .lineStyle(StrokeStyle(lineWidth: 2))
            }

            // Betting Line
            RuleMark(x: .value("Line", line))
                .foregroundStyle(.yellow)
                .lineStyle(StrokeStyle(lineWidth: 2, dash: [5, 3]))
                .annotation(position: .top, alignment: .center) {
                    Text("Line: \(line, specifier: "%.1f")")
                        .font(.caption)
                        .foregroundColor(.yellow)
                        .padding(4)
                        .background(.ultraThinMaterial)
                        .cornerRadius(4)
                }

            // Mean Line
            RuleMark(x: .value("Mean", result.meanTotal))
                .foregroundStyle(.green)
                .lineStyle(StrokeStyle(lineWidth: 1))
                .annotation(position: .bottom, alignment: .center) {
                    Text("Mean: \(result.meanTotal, specifier: "%.1f")")
                        .font(.caption2)
                        .foregroundColor(.green)
                }
        }
        .chartXScale(domain: (result.percentiles.p5 - 10)...(result.percentiles.p95 + 10))
        .chartYAxis(.hidden)
        .chartXAxis {
            AxisMarks(position: .bottom) { value in
                AxisValueLabel()
                    .foregroundStyle(.secondary)
            }
        }
        .frame(height: 200)
    }

    // Generate distribution curve points
    var distributionPoints: [(total: Double, probability: Double)] {
        let mean = result.meanTotal
        let std = result.stdTotal
        var points: [(Double, Double)] = []

        for i in stride(from: mean - 3*std, through: mean + 3*std, by: 2) {
            let z = (i - mean) / std
            let p = exp(-0.5 * z * z) / sqrt(2 * .pi)
            points.append((i, p))
        }

        return points
    }
}
```

## Summary

This wireframe mockup document provides a comprehensive blueprint for implementing the Stat Discute iOS app with:

1. **5 Main Tabs**: Today, Players, Analysis, My Bets, Settings
2. **iOS 26 Liquid Glass Design**: Glass effect containers, floating tab bar
3. **Swift Charts**: Distribution curves, performance trends, stat bars
4. **Rich Interactions**: Haptics, gestures, animations
5. **Offline-First Architecture**: Core Data caching with background sync
6. **Complete Component Library**: Reusable, well-documented components

The design maintains parity with the web application while leveraging native iOS 26 patterns for an optimal mobile experience.
